import React, { useState } from 'react';

const TextInputExample = () => {
  // input에 입력될 텍스트를 관리하는 상태 변수
  const [inputText, setInputText] = useState('');

  // 버튼 클릭 시 실행될 함수
  const handleButtonClick = (text) => {
    setInputText(text); // 버튼 클릭 시 inputText 상태 업데이트
  };

  return (
    <div>
      <input
        type="text"
        value={inputText}
        onChange={(e) => setInputText(e.target.value)} // 사용자가 직접 입력할 때도 상태 업데이트
        placeholder="텍스트가 여기에 입력됩니다"
      />
      
      {/* 버튼을 클릭할 때마다 해당 텍스트가 input에 입력됨 */}
      <button onClick={() => handleButtonClick('Hello World!')}>
        "Hello World!" 입력
      </button>
      <button onClick={() => handleButtonClick('React is awesome!')}>
        "React is awesome!" 입력
      </button>
    </div>
  );
};

export default TextInputExample;
